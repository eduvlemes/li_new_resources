API Pública Disponível

// Adicionar nova variação window.CheckoutVariationValidator.addVariation('NOVO MODELO'); // Remover variação window.CheckoutVariationValidator.removeVariation('IPHONE AIR'); // Verificar variações atuais no carrinho const variations = window.CheckoutVariationValidator.getCurrentVariations(); console.log(variations); // Limpar cache da IA window.CheckoutVariationValidator.clearIACache(); // Recarregar variações da IA (forçar recarga) await window.CheckoutVariationValidator.reloadVariationsFromIA(true); // Atualizar configuração window.CheckoutVariationValidator.updateConfig({ debug: true, colors: { confirmButton: '#ef4444' } }); // Ativar/Desativar window.CheckoutVariationValidator.activate(); window.CheckoutVariationValidator.deactivate(); // Reiniciar plugin window.CheckoutVariationValidator.reinit(); // Destruir plugin window.CheckoutVariationValidator.destroy();
