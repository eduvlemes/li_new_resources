Produtos personalizáveis

Clique em Comprar em qualquer produto. O modal de personalização será exibido. Ao confirmar, os dados são salvos no sessionStorage com a chave [produto-id]_properties e o clique no botão prossegue normalmente.

Tipos de campo suportados

Configure o array fields com qualquer combinação dos tipos abaixo:

Campos de texto
text textarea number email tel
Campos de seleção
select radio checkbox

Como usar

Defina window.ProductCustomizerConfig antes de carregar o script. O plugin intercepta automaticamente qualquer botão com a classe .botao.botao-comprar.principal. 

<!-- 1. Configuração (definida antes do script) -->
<script>
window.ProductCustomizerConfig = {
    modalTitle: "Personalize seu produto",
    modalSubtitle: "Preencha antes de adicionar ao carrinho.",
    confirmButtonText: "Confirmar e adicionar",
    cancelButtonText: "Cancelar",

    // Prefixo opcional: chave final = storagePrefix + productId + "_properties"
    storagePrefix: "",

    fields: [
        // Texto simples
        { id: "nome",     label: "Nome para gravação",   type: "text",     required: true,  placeholder: "Ex: João da Silva" },

        // Área de texto
        { id: "mensagem", label: "Mensagem especial",    type: "textarea", required: false, placeholder: "Mensagem opcional" },

        // Lista suspensa
        { id: "tamanho",  label: "Tamanho",              type: "select",   required: true,
          options: ["P", "M", "G", "GG"]
          // ou: options: [{ value: "p", label: "Pequeno" }, ...]
        },

        // Escolha única (radio)
        { id: "cor",      label: "Cor preferida",        type: "radio",    required: true,
          options: [{ value: "preto", label: "Preto" }, { value: "branco", label: "Branco" }]
        },

        // Múltipla escolha (checkbox)
        { id: "extras",   label: "Adicionais",           type: "checkbox", required: false,
          options: ["Embalagem presente", "Cartão personalizado"]
        },

        // Checkbox único (aceite)
        { id: "aceite",   label: "Confirmação",          type: "checkbox", required: true,
          placeholder: "Confirmo que revisei as informações",
          errorMessage: "Você precisa confirmar antes de continuar."
        }
    ],

    colors: {
        confirmBackground: "#2563eb",
        confirmHover: "#1d4ed8"
    }
};
</script>

<!-- 2. Script do plugin -->
<script src="resources/product_customizer.js"></script>

<!-- 3. Estrutura HTML esperada na página de produto -->
<div class="acoes-produto" data-produto-id="123">
    <button class="botao botao-comprar principal">Comprar</button>
</div>

API pública — window.ProductCustomizer

// Abre o modal manualmente
window.ProductCustomizer.open();

// Fecha o modal
window.ProductCustomizer.close();

// Ativa ou desativa o plugin sem removê-lo
window.ProductCustomizer.updateConfig({ active: false });

// Remove o plugin do DOM
window.ProductCustomizer.destroy();

// Reinicializa o plugin
window.ProductCustomizer.reinit();

// Lendo os dados salvos no sessionStorage:
const dados = JSON.parse(sessionStorage.getItem("1001_properties"));
// Retorno exemplo:
