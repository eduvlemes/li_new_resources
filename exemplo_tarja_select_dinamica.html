stripe Select Dinâmica

A barra no topo muda de texto de acordo com a região selecionada no menu dropdown. A seleção é persistida via localStorage e restaurada automaticamente.

Exemplo de configuração

window.stripeSelectConfig

<script>
window.stripeSelectConfig = {
    active: true,
    position: 'top',            // 'top' ou 'bottom'
    defaultText: 'Selecione sua região para ver a condição de frete',
    selectPlaceholder: 'Sua região ▾',

    items: [
        {
            value: 'sao-paulo',
            label: 'São Paulo',
            text: 'Adicione <b>R$ 99,99</b> ao carrinho para ter <b>FRETE GRÁTIS</b>'
        },
        {
            value: 'outros',
            label: 'Outros estados',
            text: 'Frete calculado no checkout'
        }
        // ... adicione quantas regiões precisar
    ],

    useLocalStorage: true,      // salva a seleção do usuário

    colors: {
        background: '#1a4a7a',
        text: '#ffffff',
        selectBackground: 'rgba(0,0,0,0.25)',
        selectText: '#ffffff',
        selectBorder: 'rgba(255,255,255,0.35)',
        selectArrow: '#ffffff'
    }
};
</script>
<script src="resources/stripe_select.js"></script>
